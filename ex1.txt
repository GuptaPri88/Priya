%macro FileNameReader(dir=); /*folder directory path*/
filename temp pipe 'dir "&dir."';
Data perm.history;
infile temp truncover;
input txt $1-1000;
length Status $11;
Filename = Scan(txt,-1," ");
if lowcase(strip(scan(Filename,-1,"."))) in ("csv","xls");
drop txt;
run;

%mend FileNameReader;
